\section{Weak form}

$$
\begin{gathered}
u_t-\sigma \Delta u=0 \\
%
\int_{\Omega} v u_t d x^3 -
\sigma \int_{\Omega} v \Delta u d x^3=0 . \\
%
\int_{\Omega} v \Delta u d x^3 =
\int_{\partial \Omega} v \nabla u \cdot d \mathbf{S}
%
- \int_{\Omega} \nabla v \cdot \nabla u d x^3 . \\
\int_{\Omega} \nabla v \cdot \nabla u d x^3 =
- \int_{\Omega} v u_t d x^3
\end{gathered}
$$
